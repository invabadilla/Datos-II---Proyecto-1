109 /******************
110 AllocateMemory
111 ******************/
112  bool CMemoryPool::AllocateMemory(const std::size_t &sMemorySize)
113  {
114  // This function will allocate *at least* "sMemorySize"-Bytes from the Operating-System.
115 
116  // How it works :
117  // Calculate the amount of "SMemoryChunks" needed to manage the requested MemorySize.
118  // Every MemoryChunk can manage only a certain amount of Memory
119  // (set by the "m_sMemoryChunkSize"-Member of the Memory-Pool).
120  //
121  // Also, calculate the "Best" Memory-Block size to allocate from the
122  // Operating-System, so that all allocated Memory can be assigned to a
123  // Memory Chunk.
124  // Example :
125  // You want to Allocate 120 Bytes, but every "SMemoryChunk" can only handle
126  // 50 Bytes ("m_sMemoryChunkSize = 50").
127  // So, "CalculateNeededChunks()" will return the Number of Chunks needed to
128  // manage 120 Bytes. Since it is not possible to divide 120 Bytes in to
129  // 50 Byte Chunks, "CalculateNeededChunks()" returns 3.
130  // ==> 3 Chunks can Manage 150 Bytes of data (50 * 3 = 150), so
131  // the requested 120 Bytes will fit into this Block.
132  // "CalculateBestMemoryBlockSize()" will return the amount of memory needed
133  // to *perfectly* subdivide the allocated Memory into "m_sMemoryChunkSize" (= 50) Byte
134  // pieces. -> "CalculateBestMemoryBlockSize()" returns 150.
135  // So, 150 Bytes of memory are allocated from the Operating-System and
136  // subdivided into 3 Memory-Chunks (each holding a Pointer to 50 Bytes of the allocated memory).
137  // Since only 120 Bytes are requested, we have a Memory-Overhead of
138  // 150 - 120 = 30 Bytes.
139  // Note, that the Memory-overhead is not a bad thing, because we can use
140  // that memory later (it remains in the Memory-Pool).
141  //
142 
143  unsigned int uiNeededChunks = CalculateNeededChunks(sMemorySize) ;
144  std::size_t sBestMemBlockSize = CalculateBestMemoryBlockSize(sMemorySize) ;
145 
146  TByte *ptrNewMemBlock = (TByte *) malloc(sBestMemBlockSize) ; // allocate from Operating System
147  SMemoryChunk *ptrNewChunks = (SMemoryChunk *) malloc((uiNeededChunks * sizeof(SMemoryChunk))) ; // allocate Chunk-Array to Manage the Memory
148  assert(((ptrNewMemBlock) && (ptrNewChunks)) && "Error : System ran out of Memory") ;
149 
150 
151  // Adjust internal Values (Total/Free Memory, etc.)
152  m_sTotalMemoryPoolSize += sBestMemBlockSize ;
153  m_sFreeMemoryPoolSize += sBestMemBlockSize ;
154  m_uiMemoryChunkCount += uiNeededChunks ;
155 
156  // Usefull for Debugging : Set the Memory-Content to a defined Value
157  if(m_bSetMemoryData)
158  {
159  memset(((void *) ptrNewMemBlock), NEW_ALLOCATED_MEMORY_CONTENT, sBestMemBlockSize) ;
160  }
161 
162  // Associate the allocated Memory-Block with the Linked-List of MemoryChunks
163  return LinkChunksToData(ptrNewChunks, uiNeededChunks, ptrNewMemBlock) ; ;
164  }

184 /******************
185 FreeChunks
186 ******************/
187  void CMemoryPool::FreeChunks(SMemoryChunk *ptrChunk)
188  {
189  // Make the Used Memory of the given Chunk available
190  // to the Memory Pool again.
191 
192  SMemoryChunk *ptrCurrentChunk = ptrChunk ;
193  unsigned int uiChunkCount = CalculateNeededChunks(ptrCurrentChunk->UsedSize);
194  for(unsigned int i = 0; i < uiChunkCount; i++)
195  {
196  if(ptrCurrentChunk)
197  {
198  // Step 1 : Set the allocated Memory to 'FREEED_MEMORY_CONTENT'
199  // Note : This is fully Optional, but usefull for debugging
200  if(m_bSetMemoryData)
201  {
202  memset(((void *) ptrCurrentChunk->Data), FREEED_MEMORY_CONTENT, m_sMemoryChunkSize) ;
203  }
204 
205  // Step 2 : Set the Used-Size to Zero
206  ptrCurrentChunk->UsedSize = 0 ;
207 
208  // Step 3 : Adjust Memory-Pool Values and goto next Chunk
209  m_sUsedMemoryPoolSize -= m_sMemoryChunkSize ;
210  ptrCurrentChunk = ptrCurrentChunk->Next ;
211  }
212  }
213  }

217 FindChunkSuitableToHoldMemory
218 ******************/
219  SMemoryChunk *CMemoryPool::FindChunkSuitableToHoldMemory(const std::size_t &sMemorySize)
220  {
221  // Find a Chunk to hold *at least* "sMemorySize" Bytes.
222  unsigned int uiChunksToSkip = 0 ;
223  bool bContinueSearch = true ;
224  SMemoryChunk *ptrChunk = m_ptrCursorChunk ; // Start search at Cursor-Pos.
225  for(unsigned int i = 0; i < m_uiMemoryChunkCount; i++)
226  {
227  if(ptrChunk)
228  {
229  if(ptrChunk == m_ptrLastChunk) // End of List reached : Start over from the beginning
230  {
231  ptrChunk = m_ptrFirstChunk ;
232  }
233 
234  if(ptrChunk->DataSize >= sMemorySize)
235  {
236  if(ptrChunk->UsedSize == 0)
237  {
238  m_ptrCursorChunk = ptrChunk ;
239  return ptrChunk ;
240  }
241 
242  }
243  uiChunksToSkip = CalculateNeededChunks(ptrChunk->UsedSize) ;
244  if(uiChunksToSkip == 0) uiChunksToSkip = 1 ;
245  ptrChunk = SkipChunks(ptrChunk, uiChunksToSkip) ;
246  }
247  else
248  {
249  bContinueSearch = false ;
250  }
251  }
252  return NULL ;
253  }

318 ******************/
319  bool CMemoryPool::LinkChunksToData(SMemoryChunk *ptrNewChunks, unsigned int uiChunkCount, TByte *ptrNewMemBlock)
320  {
321  SMemoryChunk *ptrNewChunk = NULL ;
322  unsigned int uiMemOffSet = 0 ;
323  bool bAllocationChunkAssigned = false ;
324  for(unsigned int i = 0; i < uiChunkCount; i++)
325  {
326  if(!m_ptrFirstChunk)
327  {
328  m_ptrFirstChunk = SetChunkDefaults(&(ptrNewChunks[0])) ;
329  m_ptrLastChunk = m_ptrFirstChunk ;
330  m_ptrCursorChunk = m_ptrFirstChunk ;
331 
332  }
333  else
334  {
335  ptrNewChunk = SetChunkDefaults(&(ptrNewChunks[i])) ;
336  m_ptrLastChunk->Next = ptrNewChunk ;
337  m_ptrLastChunk = ptrNewChunk ;
338  }
339 
340  uiMemOffSet = (i * ((unsigned int) m_sMemoryChunkSize)) ;
341  m_ptrLastChunk->Data = &(ptrNewMemBlock[uiMemOffSet]) ;
342 
343  // The first Chunk assigned to the new Memory-Block will be
344  // a "AllocationChunk". This means, this Chunks stores the
345  // "original" Pointer to the MemBlock and is responsible for
346  // "free()"ing the Memory later....
347  if(!bAllocationChunkAssigned)
348  {
349  m_ptrLastChunk->IsAllocationChunk = true ;
350  bAllocationChunkAssigned = true ;
351  }
352  }
353  return RecalcChunkMemorySize(m_ptrFirstChunk, m_uiMemoryChunkCount) ;
354  }
